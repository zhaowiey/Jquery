
这是一个ajax通常的做法，把业务逻辑都放在done中做处理

```javascript
$.ajax({
    url: "test.html",
    context: document.body
}).done(function (data) {
    //data的数据处理
    $('aaron1').html(data.a);
    $('aaron2').html(data.b);
    $('aaron3').html(data.c);
    //其余处理
});
```

改进版,把业务逻辑分离出来。

```javascript
$.ajax({
    url: "test.html",
    context: document.body
}).done(function (data) {
	pocessData();
	pocessHtml();
	pocessOther();
});

function pocessData() {
	// ...code...
}

function pocessHtml() {
	$('aaron1').html(data.a);
    $('aaron2').html(data.b);
    $('aaron3').html(data.c);
}

function pocessOther() {
	// ...code...
}
```

但以上的代码也不够理想，达不到抽象复用。
使用观察者模式进行改善

```javascript
//订阅
Observable.add(function () {
	//pocessData
});

Observable.add(function () {
	$('aaron1').html(data.a);
    $('aaron2').html(data.b);
    $('aaron3').html(data.c);
});

Observable.add(function () {
	//pocessOther
});

$.ajax({
    url: "test.html",
    context: document.body
}).done(function (data) {
	//发布
	Observable.fire(data);
});
```

一个完整的小例子

```javascript
//定义一个简易的观察者模型
var Observable = {
	callbacks: [],

	//订阅方法
	add: function (fn) {
		this.callbacks.push(fn);
	},

	//发布方法
	fire: function (data) {
		this.callbacks.forEach(function (fn) {
			fn(data);
		});
	}
};

//模拟一个ajax
function ajax(arg) {
	var data = arg.data,
		fn = arg.success;
	setTimeout(function () {
		fn("返回数据 " + data);
	}, 1000);
}

//订阅
Observable.add(function (data) {
	console.log('func1 ' + data);
});

Observable.add(function (data) {
	console.log('func2 ' + data);
});

ajax({
	data: 'some data',
	success: function (data) {
		//发布
		Observable.fire(data);
	}
});

//输出
//func1 返回数据 some data
//func2 返回数据 some data
```

总的来说，观察者模式所做的工作就是在解耦，让耦合的双方都依赖于抽象，而不是依赖于具体。
